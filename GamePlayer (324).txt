					if (firstLetterUppercase)
						return "She";
					else
						return "she";
				case 1:
					// Possessive
					if (firstLetterUppercase)
						return "Her";
					else
						return "her";
				case 2:
					// Objective
					if (firstLetterUppercase)
						return "Her";
					else
						return "her";
			}
		}
