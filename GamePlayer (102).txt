					return GlobalConstants.REALM_RANK_NAMES[(int)Realm - 1, (int)Gender - 1, (RealmLevel / 10)];
				}
				catch
				{
					return "Unknown Rank"; // why aren't all the realm ranks defined above?
				}
			}
		}
