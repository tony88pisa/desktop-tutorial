					InventoryItem item = m_player.Inventory.GetItem(eInventorySlot.Horse);
					if (item != null)
						m_level = item.Level;
					else
						m_level = 35;//base horse by default
					m_player.Out.SendSetControlledHorse(m_player);
				}
			}
